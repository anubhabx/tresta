JavaScript Fetch Example

<script>
// Configuration
const WIDGET_ID = 'YOUR_WIDGET_ID';
const API_URL = 'http://localhost:8000'; // or https://api.tresta.io in production

// Fetch widget data
async function loadWidget() {
    try {
        const response = await fetch(`${API_URL}/widgets/${WIDGET_ID}/public`);

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const result = await response.json();

        if (result.success) {
            renderWidget(result.data);
        } else {
            throw new Error(result.error?.message || 'Failed to load widget');
        }
    } catch (error) {
        console.error('Widget load error:', error);
        showError(error.message);
    }
}

// Render testimonials
function renderWidget(data) {
    const container = document.getElementById('widget-content');
    const testimonials = data.testimonials || [];

    if (testimonials.length === 0) {
        container.innerHTML = '<p>No testimonials yet.</p>';
        return;
    }

    container.innerHTML = testimonials.map(t => `
        <div class="testimonial">
            ${t.rating ? `<div class="testimonial-rating">${'‚≠ê'.repeat(t.rating)}</div>` : ''}
            <div class="testimonial-content">"${t.content}"</div>
            <div class="testimonial-author">
                ${t.authorImage ? `<img src="${t.authorImage}" class="testimonial-image">` : ''}
                <div class="testimonial-info">
                    <div class="testimonial-name">${t.authorName}</div>
                    ${t.authorRole ? `<div class="testimonial-role">${t.authorRole}</div>` : ''}
                </div>
            </div>
        </div>
    `).join('');
}

// Load widget on page load
loadWidget();
</script>

    $.ajax({
        url: `${API_URL}/widgets/${WIDGET_ID}/public`,

            if (result.success && result.data.testimonials) {
                renderTestimonials(result.data.testimonials);
            }
        },
        error: function(xhr, status, error) {
            console.error('Widget load failed:', error);
            $('#widget-content').html('<p>Failed to load testimonials.</p>');
        }
    });
});

React Example

import { useEffect, useState } from 'react';

function TrestaWidget({ widgetId }) {
    const [testimonials, setTestimonials] = useState([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);

    useEffect(() => {
        async function fetchWidget() {
            try {
                const response = await fetch(
                    `http://localhost:8000/api/widgets/${widgetId}/public`
                );
                const result = await response.json();

                if (result.success) {
                    setTestimonials(result.data.testimonials || []);
                } else {
                    throw new Error(result.error?.message);
                }
            } catch (err) {
                setError(err.message);
            } finally {
                setLoading(false);
            }
        }

        fetchWidget();
    }, [widgetId]);

    if (loading) return <div>Loading testimonials...</div>;
    if (error) return <div>Error: {error}</div>;

    return (
        <div className="tresta-widget">
            {testimonials.map((testimonial) => (
                <div key={testimonial.id} className="testimonial">
                    {testimonial.rating && (
                        <div className="rating">
                            {'‚≠ê'.repeat(testimonial.rating)}
                        </div>
                    )}
                    <p>{testimonial.content}</p>
                    <div className="author">
                        {testimonial.authorImage && (
                            <img src={testimonial.authorImage} alt={testimonial.authorName} />
                        )}
                        <div>
                            <strong>{testimonial.authorName}</strong>
                            {testimonial.authorRole && <p>{testimonial.authorRole}</p>}
                        </div>
                    </div>
                </div>
            ))}
        </div>
    );
}

export default TrestaWidget;

CORS Configuration

The Tresta API is configured to allow requests from any origin for public widget endpoints:

// Public widget endpoint CORS configuration
{
    origin: "*",                    // Allow any origin
    methods: ["GET", "OPTIONS"],    // Read-only access
    allowedHeaders: ["Content-Type"],
    credentials: false,             // No authentication required
    maxAge: 86400                   // Cache preflight for 24 hours
}
Security Note: Public widget endpoints are read-only and only return approved testimonials from public projects. No sensitive data or authentication is required, making CORS wildcard safe.
